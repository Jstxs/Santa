    <h1>Secret Santa Picker</h1>

    <p>Click the button below to shuffle and assign Secret Santa pairs!</p>

    <button onclick="pickSecretSanta()">Pick Secret Santa</button>

    <script>

        const participants = ["Justin", "Jaryl", "Kyle", "Yuma", "Mika", "Nicole", "Alina", "Yosei", "Andrei", "Ceejay", "Kazuya", "Jones", "Jun", "Versie"];


        function shuffle(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

            return array;

        }


        function pickSecretSanta() {

            const remainingReceivers = [...participants];

            const assignments = participants.map(giver => {

                const possibleReceivers = remainingReceivers.filter(receiver => receiver !== giver);

                const randomIndex = Math.floor(Math.random() * possibleReceivers.length);

                const receiver = possibleReceivers[randomIndex];


                // Remove the selected receiver from the pool

                remainingReceivers.splice(remainingReceivers.indexOf(receiver), 1);


                return `${giver} → ${receiver}`;

            });


            const resultDiv = document.getElementById('result');

            resultDiv.innerHTML = `<h2>Secret Santa Assignments:</h2><ul>${assignments.map(a => `<li>${a}</li>`).join('')}</ul>`;

            resultDiv.classList.remove('hidden');

        }

    </script>
